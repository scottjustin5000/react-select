## Install

```bash
npm install --save react-select
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-select'
import 'react-select/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
